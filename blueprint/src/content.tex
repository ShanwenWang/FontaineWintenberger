\section{Introduction}

\begin{definition}
    Let $K$ be a local field and $L/K$ be an extension. We say the extension $L/K$ is arithmetically profinite if for all $u\geq -1$, we have $G_K^uG_L$ is open in $G_K$. 
\end{definition}
